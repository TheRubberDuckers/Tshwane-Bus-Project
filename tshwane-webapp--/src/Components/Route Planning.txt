line = null
let active = false



function draw_route (data, map, active, colors):
	//Create the line once if  it doesnt exist
	if (!line) {
		line = L.polyline(data,{color: colors} )
	}	
	if (active) {
		line.addTo(map)
	}else {
		map.removeLayer(line)
	}
